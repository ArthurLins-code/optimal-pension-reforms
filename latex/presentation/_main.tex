        \item Where $\eta$ is the average welfare weight on beneficiaries of the reform

        \item For budget-neutral, WPIC = WMVPF $\Rightarrow$ increase benefit slope and decrease benefit level
        %\item (Preliminary) Optimal reform consists of increasing benefit slope and decreasing benefit level
        \item For expenditure financed by payroll tax, WPIC = WNSBD $\Rightarrow$ increase benefit slope and finance it with the payroll tax
    \end{itemize}
\end{itemize}

\begin{frame}[label=lengthService]{Length-of-Service Pension Schedule}
    \begin{itemize}

        \item Benefits capped by a floor (minimum wage) and a ceiling (5.6 minimum wages $\approx$ 90th percentile of the income distribution)

\section{Model}

\roadmapframe{2}

\begin{frame}[label=model]
\frametitle{Agent's problem}
\begin{itemize}
    \item Agents maximize their expected utility s.t. a budget constraint
    \begin{align*}
        \max_{x_{it}} &\Bigg\{\sum_{s=t}^{T_i}\beta^s E[ u(x_{is})]\quad \text{ s.t. } \\
         &\quad a_{i,t+1} =(1+r)[a_{it}+y_{it}(x_{it})-\tau(x_{it})+b(x_{it})-c_{it}]\Bigg\}
    \end{align*}
    \item Choice vector $x_{it}$ includes assets $a_{it}$, consumption $c_{it}$, whether to claim benefits, formal and informal work, effort, etc.
    \item They carry to the next period all income $y_{it}$, assets $a_{it}$ and benefits $b$ that are not used to pay taxes $\tau$ or consumption $c_{it}$
    % b(x_it) should actually be b_t({x_is}s=0^t) but we are saving notation.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Welfare and Government's Budget}
\begin{itemize}
    \item In the reform period $t=0$, welfare is $\sum_{i} \sum_{t=0}^T\beta^t E[ u(x_{it})]$
    \item The government budget constraint (GBC) is
    \begin{align*}
        \sum_{t=0}^T \sum_i \frac{E[\tau(x_{it})]-E[b(x_{it})]}{(1+r)^t}\geq\bar G_0,
    \end{align*}
    % This is weaker than requiring that the budget closes under each state of the world.
    \item Two benefit schedule concepts
    % technically, it also depends on states, but we are saving notation
    % Note that the actual schedule differs for those who claimed above the threshold depending on whether they claim before or after the reform
    % The counterfactual schedule is the same
    \begin{itemize}
        \item $b^c(\cdot)$: Counterfactual schedule in the absence of the reform
        \item $b^a(\cdot)$: Actual schedule with the reform
    \end{itemize}
    \item They lead to two different choice-vector concepts
    \begin{itemize}
        \item $x^c_{it}$: under the counterfactual schedule
        \item $x_{it}^a$: under the actual (reformed) schedule
    \end{itemize}
    %\item These choices also may affect the state distribution
\end{itemize}
\end{frame}

\begin{frame}{Welfare Analysis}
    \begin{itemize}
        \item By the envelope theorem, the welfare effect per dollar spent in the reform from $b^c$ to $b^a$ is:
        \begin{align*}
            WMVPF(b^c,b^a)=\frac{\sum_t \sum_i E\left(\beta^t \frac{\partial u(x_{it})}{\partial c_{it}}[b^a(x^c_{it})-b^c(x^c_{it})]\right)}{C(b^c,b^a)}
        \end{align*}
        % b_a'=b^a(x_{it}|a(x_{it})=a)
        % b_a=b(x_{it}|a(x_{it})=a)
        \item Implicit assumption: utilitarian planner
        \item $C(b^c,b^a)$ is the cost of the reform, including fiscal externalities:
    \end{itemize}
    \small
    \begin{align*}
        \small
        \sum_{t=0}^T \sum_i \frac{\overbrace{E[b^a(x_{it}^a)]-E[b^c(x_{it}^c)]}^{\text{Benefit Changes}}-
        \overbrace{\left[E[\tau(x_{it}^a)]-E[\tau(x_{it}^c)]\right]}^{\text{Revenue Changes}}}{(1+r)^t}
    \end{align*}
\end{frame}

\begin{frame}[label=MEandFE]{Mechanical Effect and Willingness-to-Pay}
    \begin{itemize}
        \item WTP is captured by mechanical effect (envelope theorem)
    \end{itemize}
    {\small
    \begin{align*}
         \sum_i \sum_{t=0}^{T_i} \frac{E[b^a(x^c_{it})]-E[b^c(x_{it}^c)]}{(1+r)^t}
    \end{align*}
    }
    \begin{itemize}
        \item Mechanical Effect: Counterfactual increase in benefit outlays under the reformed schedule but in the absence of behavioral responses
        \item Can recover $b^a(\mathbf{x_{it}^a})$ and $b^c(\mathbf{x_{it}^a})$ but need $b^a(\mathbf{x_{it}^c})$ and $b^c(\mathbf{x_{it}^c})$
        % \begin{enumerate}
        %     \item counterfactual benefits $\sum_t b^c(\mathbf{x_{it}^c})/(1+r)^t$
        %     \item mechanical benefits $\sum_t b^a(\mathbf{x^c_{it}})/(1+r)^t$
        % \end{enumerate}
        \item With the counterfactual claiming frequencies $N^c_{p,t}$, we just need the avg counterfactual and mechanical benefits at each normalized point
        \begin{align*}
            \bar b^c_{p,t} \equiv \frac{1}{N_{p,t}^c}\sum_{i:p(x_{it})=p}\sum_t \frac{b^c(x_{it}^c)}{(1+r)^t}
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}[label=avgCounterfactualBenefits]{Average Counterfactual Benefits}
% This should be the sum of the PDV (from the perpective of the refomr) of benefits for those who claim at t with p points from t=0 to T_i.
% So p(x_it) = p only if i claims at t with p.
    \begin{itemize}
        \item Only measure $\bar{b}^{c,a}_{p,t} \equiv \frac{1}{N_{p,t}} \sum_{i:p(x_{it}^a)=p}\sum_t b^c(x_{it}^a)/(1+r)^t$
        \item Problem: \textit{Bunchers} could be positively selected (eg, higher wages)
        \item Identification Assumption: $\bar{b}^{c,a}_{p,t}$ for $p\geq -6$ would remain parallel to the ones in $p<-6$ in the absence of the reform
        \item The DiD recovers the effect of the reform on these average benefits $\beta^{c,a}_{k,p}$ at each normalized point $p$ and quarter $k$:
         \begin{align*}
            \bar{b}^{c,a}_{p,t}=\delta_p^{c,a}+\gamma_t^{c,a} +\sum_{k\neq -2}\beta_{p,k}^{c,a} 1(p\in \tilde p)*1(t=k)+\varepsilon_{p,t}^{c,a}
        \end{align*}
        \item $\bar{b}^{c,a}_{p,t}-\hat \beta_{p,t}^{c,a}$ is an estimate for $\bar b^c_{p,t}$
    \end{itemize}
    \bfootnote{\hyperlink{npv}{\beamerbutton{PDV Details}}}
    % The estimate for the average benefit that would have been paid in the absence of the reform at point p and quarter t is estimated as bar b^o_{p,t}-\beta_{t,p}
\end{frame}

\begin{frame}[label=costReform]{Cost of the Reform}

\begin{frame}[label=wmvpfMvpf]{WMVPF and MVPF of the Reform}
    \begin{itemize}
        \item Under envelope conditions, the welfare effect per dollar spent is:
        % It is an approximation because we are considering a non-infinitesimal reform
        \begin{align*}
            WMVPF(b^c,b^a)&\approx \frac{\sum_i \sum_t E\left(\beta^t \frac{\partial u(x_{it})}{\partial c_{it}}*[b^a(x_{it}^c)-b^c(x_{it}^c)]\right)}{C(b^c,b^a)} \\
            & = \eta \frac{\sum_i \sum_t  E[b^a(x_{it}^c)-b^c(x_{it}^c)]}{C(b^c,b^a)} =\eta * MVPF(b^c,b^a)
        \end{align*}
        % technically, we are also holding constant the state of the world in b^a(x,pi). I believe we want the same distribution as in the absence of the reform.
        % For example, if someone gets fired with a higher probability under the reformed schedule, we want to compute their benefits with the lower probability before the reform was enacted
        \item Where $\eta$ is the incidence-weighted average welfare weight
        \item Normalize mgU to the avg. in the economy + constant CRRA $\gamma=4$:
        %\equiv \frac{\sum_i \sum_t E\left(\beta^t \frac{\partial u(x_{it})}{\partial c_{it}}*[b^a(x_{it}^c)-b^c(x_{it}^c)]\right)}{\sum_i \sum_t E\left[b^a(x_{it}^c)-b^c(x_{it}^c)\right]}
        % CRRA constant across individuals, allow us to write gamma without i
        % CRRA constant across consumption levels allow us to use equal instead of approx
        \begin{align*}
            \eta = \beta^{t}\left(1-\gamma\frac{c_{p}-c}{c}\right)
        \end{align*}
        \begin{itemize}
            \item $c_p$ is the avg. cons. among beneficiaries, and $c$ is avg consumption in Brazil $\Rightarrow \frac{c_{p}-c}{c}=4.2\%$ from surveys
        \end{itemize}
    \end{itemize}

\begin{frame}[label=theory]{Two-Step Agent Problem}
    \begin{align*}
        \tilde{U}_{i0}(p,b) =\max_{x_{i0}} \Big\{\sum_{t=0}^T\beta^t E[u(x_{it})]\ \text{s.t. the Budget Constraint}, \\
         p(x^{iT})=p,\ \text{ and }\ b(x_{it})=b\Big\}
    \end{align*}
    \begin{itemize}
        \item $\tilde{U}_{i0}(p,b)$ reduced-form expected utility in terms of points and benefits given rational choices.
        \item Imply indifference curves decreasing in $\beta$ ($-\frac{d}{d\beta}\frac{d\tilde U/dp}{d\tilde U/db}<0$)
        \item Two sets of agents:
        \begin{enumerate}
            \item Unconstrained agents: Get to reform with $p(x_{i0})<\bar p$ (can bunch at $\bar p$)
            \item Constrained agents: Get to reform with more than $p(x_{i0})>\bar p$ points (cannot bunch and can only choose between $[p(x_{i0}),\infty)$)
        \end{enumerate}
    \end{itemize}
    \bfootnote{\hyperlink{twoCounterfactualReforms}{\beamerbutton{Back}}}
\end{frame}

\begin{frame}{\textcolor{red}{Pure Reforms Frequencies: Announcement ($t=-1$)}}
\begin{itemize}
    \item For $p<0$, there is only postponement that would  occur under pure-L, but not under the pure-S reform. Hence:
    \[
    N^L_{p,-1}=N^a_{p,-1}
    \qquad\text{and}\qquad
    N^S_{p,-1}=N^c_{p,-1}
    \]
    \item For $p\geq 0$, actual reform induces postponement at $t=-1$ due to larger post-reform benefits (as in the pure-L reform)
    \item Hence, for already-eligible claimants:
    \[
    N^L_{p,-1}=N^a_{p,-1}, \qquad p\geq 0
    \]
\end{itemize}
\end{frame}

\begin{frame}{\textcolor{red}{Pure-S Anticipation at $t=-1$: Logic}}
\begin{itemize}
    \item Under the pure-S reform, the level increase is absent and waiting becomes less attractive for $p>0$
    \item Thus, already-eligible claimants who delay under the actual reform would instead anticipate under the pure-S reform
    \item A1 (Attention Mapping): if an individual reacts with delay $x$ quarters after implementation under the actual reform, she would react with delay $x$ quarters after announcement under the pure-S reform
    \begin{itemize}
        \item This implies that early excess mass under the actual reform in $t$ is shifted to $t-1$ under the pure-S reform
    \end{itemize}
    \item Because points rise by $0.5$ per quarter while $p$ is binned at integers, mass from $(p,0)$ is splitted back to $(p,-1)$ and $(p-1,-1)$ for $p\geq 0$
\end{itemize}
\end{frame}

\begin{frame}{\textcolor{red}{Pure-S Frequencies at $t=-1$ and $p\geq 0$}}
    \begin{itemize}
        \item Define early excess mass in cell $(p,t)$ as $EM_{p,t}\equiv N^a_{p,t}-N^c_{p,t}$
        \item A1 implies that incoming anticipation mass at $p\geq 0$ and $t=-1$:
        \[
        A_{p,-1}\equiv \frac12 EM_{p,0}+\frac12 EM_{p+1,0}.
        \]
        \item For $p\geq 0$ and $t=-1$, the frequency under the pure-S reform equals the counterfactual distribution (as nobody would postpone) + this anticipation mass (as anticipators would claim before the reform):
        \begin{align*}
            N^S_{p,-1}=N^c_{p,-1}+A_{p,-1}
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}{Assumptions: Point Accumulation and Bunching Window}
\begin{itemize}
    \item A3: Bunching only happens in $p\in[0,\textcolor{red}{3})$, corresponding to responses within \textcolor{red}{3} quarters after first facing the incentive
    \item In the first post-reform quarters, excess mass can appear beyond $p=2$ because anticipation by already-eligible claimants is diffuse
    \item The key empirical restriction is that this higher-$p$ excess largely disappears after $t=\textcolor{red}{3}$
    \item This motivates using early excess mass only for $t\in\{0,1,2,3\}$ when constructing pure-S anticipation
\end{itemize}
\end{frame}

\begin{frame}[label=pureL]{Pure-L Frequencies for $t\geq 0$: Arrivals at Cutoff}
\begin{itemize}
    \item Goal: Get \emph{pure level} frequencies $N^{L}_{p,t}$
    \item Number of postponers from $(p,s)$: $P_{p,s}\equiv N^{c}_{p,s}-N^{a}_{p,s}\quad\text{for }p<0$
    \item A2: Points accumulate deterministically at $0.5$ per quarter
    \item For $p<0$, postponer from $(p,s)$ reaches cutoff $p=0$ at $s-2p\equiv t$, so that \emph{postponement arrivals at the cutoff} in quarter $t$ are:
    \[
        PA_{t} \equiv \sum_{p\in\{-6,-5.5,\dots,-0.5\}} P_{p,\;t+2p},
    \]
    \item Interpretation: $PA_{t}$ is the number of postponers that first become able to claim with $p=0$ (arrive at the threshold) in quarter $t$
\end{itemize}
\end{frame}

\begin{frame}{Pure-L Frequency}
    \begin{itemize}
        \item A2 implies that a claim in $(p,t)$ must come from arrivals at $t-2p$
        \begin{itemize}
            \item Ex: the only postponers who can be at $(p,t)=(1,4)$ are those that arrived in $t=2$ and then claimed 2 quarters later
        \end{itemize}
        \item Postponement bunching at $(p,t)$ is arrivals in $t-2p$ ($PA_{t-2p}$) times the probability of claiming with $p$ given arrival in $t-2p$ ($g_{p,t-2p}$):
    \[
        PB_{p,t}\equiv g_{p,t-2p}\cdot PA_{t-2p}
    \]
    \item Post pure-L reform frequency:
    \[
        N^{L}_{p,t}=
        \begin{cases}
            N^{a}_{p,t}, & p<0,\\[3pt]
            N^{c}_{p,t}+PB_{p,t}, & p\ge 0.
        \end{cases}
    \]
\end{itemize}
\end{frame}

\begin{frame}[label=gp_est]{Estimating $g_{p,t}$ from Arrival Cohorts (Counts)}
\begin{itemize}
    \item A3: Bunching only happens for \textcolor{red}{$p=0,1$}, which corresponds to delays in claiming up to \textcolor{red}{2} quarters after reaching the threshold, justified by:
        \begin{itemize}
            \item In the first periods, excess mass could occur for non-bunching anticipation (which also only appears up to \textcolor{red}{$t=3$})
            % Suggesting that the less attentive ppl were inattentive for 3 quarters
            % If someone reaches p=0 and only responds 3 quarters later it will claim with 1.5 points
            \item In later periods $(t\geq \textcolor{red}{4})$, excess mass is only observed for \textcolor{red}{$p=0,1$}
            % When there is only actual bunching incentives, it is diffuse up to 4
        \end{itemize}
    \item A4 (Proportional Mixing): Conditional on arriving on $t$, anticipators and postponers claim proportionally across \textcolor{red}{$p=0,1$}. Note that
    \begin{itemize}
        \item Even those who arrive at $t=0$ may claim  at $(p,t)=(1,2)$.
        \item We consider who arrives at $t\geq 0$, but not anticipators already above the cutoff at the reform
    \end{itemize}
    \item Empirical claiming distribution conditional on arrivals in $t>0$:
    \[
        \widehat{ g}_{p,t}\equiv
        \frac{\#\{i:\text{ arrived in } t\text{, and claimed with }p\}}
             {\#\{i:\text{ arrived in } t\text{, and claimed with }\textcolor{red}{p\in [0,1]}\}},
        \quad \textcolor{red}{p=0,1}
    \]
\end{itemize}

\end{frame}

\begin{frame}[label=pureS]{Post Pure-S Reform Frequencies}
    \begin{itemize}
        \item Remember that $EM_{p,t}\equiv N^a_{p,t}-N^c_{p,t},
        \quad p\geq 0,\; t\in\{0,1,2,3\}$, and let $EM_{p,t}=0$ for $t>3$.
        % This is because the excess mass we are talking about is only for early quarters anticipation
        % The later quarters anticipation consist of individuals that had not arrived by t=0, would have claim at p>0 but anticpate to p=0. They would do the exact same thing under the pure-S reform.

        \item Define incoming anticipation mass:
        \[
        A_{p,t}\equiv \frac12 EM_{p,t+1}+\frac12 EM_{p+1,t+1}.
        \]
        \item Then, for $p\geq 0$,
        \[
        N^S_{p,t}=N^c_{p,t}+A_{p,t}-PB_{p,t}.
        \]
         \item Since $EM_{p,t}=0$ for $t>3$, this automatically shuts down the anticipation adjustment for later periods, so no extra third line is needed.
    \end{itemize}
    \bfootnote{\hyperlink{pureLStep}{\beamerbutton{Back}} }
\end{frame}

\begin{frame}{Post Pure-S Reform Frequencies}
    \begin{itemize}
        \item Pure-S reform would create all the movements in the actual reform above the cutoff except for the postponement:
    \end{itemize}
    \begin{align*}
            N^S_{p,t}&=\begin{cases}
                N_{p,t}^c, & if\quad p<0 \\
                N_{p,t}^a - PB_{p,t}, & if\quad p\geq 0
            \end{cases}
        \end{align*}
    \begin{itemize}
        \item Since $PB_{p,t}=0$ outside $p\in[0,4)$, this subtraction only affects cells where postponement bunching can arise
    \end{itemize}
\end{frame}

\begin{frame}[label=pureDetails]{Overview of Pure Reform Computation}
\begin{itemize}
    \item Choices (e.g., claiming, years of contribution, wage, etc.) for individual $i$ in quarter $t$ are denoted by $x_{it}$
    \begin{itemize}
        \item $x^a_{it}$: Actual choices observed after the reform
        \item $x^c_{it}$: Counterfactual choices in the absence of the reform
        \item $x^L_{it}$: Choices that would have been made under the pure-L reform
    \end{itemize}
    \item Goal: construct 3 analogous benefit objects under each pure reform
    \begin{itemize}
        \item $b^{L}(x^a_{it})$: benefits applying $b^{L}$ to observed post-reform choices $x^a_{it}$
        \item $b^{L}(x^c_{it})$: mechanical benefits paid without behavioral responses
        \item $b^{L}(x^{L}_{it})$: benefits under the pure-L schedule using only pure-L behavioral responses
    \end{itemize}
    \item Same logic applies for the pure slope reform ($b^{S}$)
\end{itemize}
\end{frame}

\begin{frame}{Step 1: Compute Actual Expenditures $E^{a,L}_{p,t}$}
\begin{itemize}
    \item Apply the pure level schedule $b^{L}$ to observed post-reform choices $x^a_{it}$
    \item We approximate this using replacement rates (RR): see next slide

    \item {Compute total expenditure in each cell $(p,t)$ as:}
    \begin{align*}
        {E^{a,L}_{p,t} \equiv \sum_{i \in \{p|t\}} \sum_{\tau=t}^{T_i} \frac{b^L(x^a_{i\tau})}{(1+r)^{\tau}}}
    \end{align*}

    \item {Note this corresponds to $N^a_{p,t}\,\bar b^L(x^a_{p,t})$ in the previous notation :}

    \item {$E^{a,L}_{p,t}$ is the PV of total expenditures with those who claim with $p$ points in quarter $t$ under the pure-L schedule and observed choices}
\end{itemize}
\end{frame}

\begin{frame}{Step 2: Mechanical Expenditure under Pure Reforms}
\begin{itemize}
    \item For $q\in\{L,S\}$, recover selection with the same cell DD:
\end{itemize}
\vspace{-0.7em}
{\small
\begin{align*}
    y^{a,q}_{p,t}
    =
    \delta_{p}^{q}
    +
    \gamma_t^{q}
    +
    \sum_{k\neq -2}\beta_{\tilde p,k}^{y,q}
    1(p\in \tilde p)1(t=k)
    +
    \varepsilon_{p,t}^{y,q}
\end{align*}}
\vspace{-1.0em}
\begin{itemize}
    \item $y$ can be either expenditure $E^{a,q}_{p,t}$ or average benefits $\bar b^{q,a}_{p,t}\equiv E^{a,q}_{p,t}/N^a_{p,t}$.
    \item Expenditure DD:
    \[
        \widehat E^{c,q}_{p,t}
        =
        E^{a,q}_{p,t}
        -
        \widehat\beta^{E,q}_{p,t}.
    \]
    \item Avg.-benefit DD plus counterfactual frequencies:
    \[
        \widehat E^{c,q}_{p,t}
        =
        \widehat N^{c}_{p,t}
        \left[
        \bar b^{q,a}_{p,t}
        -
        \widehat\beta^{\bar b,q}_{p,t}
        \right].
    \]
\end{itemize}
\end{frame}

\begin{frame}{Step 2.1: No-Reform Counterfactual Expenditure}
\begin{itemize}
    \item Expenditures under the no-reform schedule $b^c$ and choices $x^c_{it}$: 
    \begin{align*}
        E^{c}_{p,t}=\sum_{i \in \{p|t\}} \sum_{\tau=t}^{T_i} \frac{b^c(x^c_{i\tau})}{(1+r)^{\tau}}
    \end{align*}
    \item Recover selection with the same cell DD:
\end{itemize}
\vspace{-0.7em}
{\small
\begin{align*}
    y^{a,c}_{p,t}
    =
    \delta_{p}^{c}
    +
    \gamma_t^{c}
    +
    \sum_{k\neq -2}\beta_{\tilde p,k}^{y,c}
    1(p\in \tilde p)1(t=k)
    +
    \varepsilon_{p,t}^{y,c}
\end{align*}}
\vspace{-1.0em}
\begin{itemize}
    \item $y$ can be either expenditure $E^{a,c}_{p,t}$ or average benefits $\bar b^{c,a}_{p,t}\equiv E^{a,c}_{p,t}/N^a_{p,t}$.
    \item Recover the no-reform counterfactual using either:
    \[
        \widehat E^{c}_{p,t}
        =
        E^{a,c}_{p,t}
        -
        \widehat\beta^{E,c}_{p,t}
        \quad\text{or}\quad
        \widehat E^{c}_{p,t}
        =
        \widehat N^{c}_{p,t}
        \left[
        \bar b^{c,a}_{p,t}
        -
        \widehat\beta^{\bar b,c}_{p,t}
        \right].
    \]
\end{itemize}
\end{frame}

\begin{frame}{Step 3: Mechanical Expenditures under Pure Level Reform}
\begin{itemize}
    \item These are the mechanical expenditures with higher benefits but no behavioral responses


    \item We aggregate {counterfactual expenditures} across normalized points:
    \[
        \text{MECH}^{L}_t
        \;=\;
        \sum_{p} {E^{c,L}_{p,t}}
    \]

    \item This is the PV of total benefit outlays under the $b^L$ reform absent any behavioral response for those claiming in $(p,t)$
\end{itemize}
\end{frame}

\begin{frame}{Step 4: Pure-L Expenditures $E^{L}_{p,t}=\sum_{i \in \{p|t\}}\sum_{\tau=t}^{T_i} \frac{b^L(x^L_{i\tau})}{(1+r)^{\tau}}$}
\begin{itemize}
    \item Since pure level reform generates {postponement} but no {anticipation}, we only need to remove selection from the latter

    \item Since only postponement affects $p<0$:
    \[
        {
        E^{L}_{p,t} \;=\; E^{a,L}_{p,t}
        }\quad \text{ for }\quad p<0,\ t\geq -1
    \]
    \item \textcolor{red}{For $p\geq 0$ and $t=-1$,  $E^{L}_{p,t}=E^{a,L}_{p,t}$ because there is postponement both in the actual and the pure-L reforms:}
    \item For $p\geq 0$ and $t\geq 0$, we construct expenditures by combining counterfactual expenditures with postponement inflows $E^{P,L}_{p,t}$:
    \[
        {
        E^{L}_{p,t}
        \;=\;
        E^{c,L}_{p,t}
        \;+\;
        E^{P,L}_{p,t}
        }
    \]
\end{itemize}
\end{frame}

\begin{frame}{Postponement Effects: Examples}
    \begin{itemize}
        \item Ex1: Postponers to $(p,t)=(1,6)$ come from $(-1,2)$ and $ (-2,0)$, but cannot come from $(-3,-2)$, $(-4,-4)$, $(-5,-6)$ nor $(-6,-8)$
        \begin{itemize}
            \item They come from $(-x,t-2(x+1))$ for $x=1,2$
            \item {All these groups arrive at the threshold in quarter $t-2p=4$, but remember they claim with $1$ with probability $g_{1,4}$}
        \end{itemize}

        \item Ex2: Postponers to $(p,t)=(2,6)$ come from $(-1,0)$, but cannot come from $(-2,-2)$, $(-3,-4)$, $(-4,-6)$, $(-5,-8)$ nor $(-6,-10)$
        \begin{itemize}
            \item They come from $(-x,t-2(x+2))$ for $x=1$
            \item {They arrive at the threshold in quarter $t-2p=2$, but remember they claim with $2$ with probability $g_{2,2}$}
        \end{itemize}

        \item {Construct postponement expenditures in $(p,t)$ from expenditure losses in origin cells}
        \begin{align*}
        \text{Ex1:}\quad
        {E^{P,L}_{1,6}}
        &{= g_{1,4}\Bigl[-E^{P,L}_{-1,2}-E^{P,L}_{-2,0}\Bigr]}
        \\
        \text{Ex2:}\quad
        {E^{P,L}_{2,6}}
        &{= g_{2,2}\Bigl[-E^{P,L}_{-1,0}\Bigr]}
        \end{align*}
        \item {Where $E^{P,L}_{p,t}=E^{a,L}_{p,t}-E^{c,L}_{p,t}$ for all $p\in (-6,0)$ and $t\geq -1$}
    \end{itemize}
\end{frame}

\begin{frame}{Postponement Effects: General Case}
    \begin{itemize}
        \item To get from $-x$ to $p$, individuals need $2x$ periods to get to zero and then $2p$ periods to get to $p$

        \item So if they got from $-x$ to $p$ in $t$, they need to have left in $t-2(x+p)$

        \item Postponers to $(p,t)$ come from $(-x,t-2(x+p))$ for $x=1,..,\bar{x}_{p,t}$

        \item  Where $-\bar x_{p,t}$ is the most negative point where postponers to $(p,t)$ came from associated with the quarter of the announcement $t-2(\bar x_{p,t}+p)=-1$ if such point is at least $-6$, i.e.
        \begin{align*}
            \bar x_{t,p}=\begin{cases}
                (t+1)/2-p & \text{ if } (t+1)/2-p\leq 6 \\
                6 & \text{ otherwise.}
            \end{cases}
        \end{align*}
        \item {Remember that all these groups arrive at the threshold in $t-2p$ and claim with $p$ with probability $g_{p,t-2p}$}
        \item {Construct total postponement expenditures in $(p,t)$ as the reallocation of expenditure losses from origin cells}
        \begin{align*}
            E^{P,L}_{p,t}
            &=
            g_{p,t-2p}
            \sum_{x=1}^{\bar x_{t,p}}
            \Bigl(
            -E^{P,L}_{-x,t-2(x+p)}
            \Bigr) \quad \text{for } p \in [0,4)
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}{Pure Level Behavioral and Counterfactual Expenditures}
    \begin{itemize}
        \item {This ensures we recover $E^{L}_{p,t}$ for all $p$}

        \item Behavioral expenditures in quarter $t$:
        \[
            {
            \text{BEHAV}^{L}_t
            \;=\;
            \sum_{p}
            E^{L}_{p,t}
            }
        \]

        \item {Combines pure-L behavioral shifting with reallocated expenditures}

        \item We can also recover the counterfactual benefit outlays as before

            \text{CNTRF}_t
            =
            \sum_{p}
            E^{c}_{p,t}
            }\equiv \sum_{p}
            N^{c}_{p,t}\,
            \bar b(x_{p,t})
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}[label=optimalDetails]{Optimal Budget-Balanced Reform Details}
    \begin{itemize}
        \item Which schedule reform increases welfare the most?
        \begin{itemize}
            \item Local in the sense that the government can only increase costs and revenues by a fixed amount set to $C(b^c,b^a)$
        \end{itemize}
        \item If the policymaker needs to balance its budget with $b_S$ and $b_L$:
        \begin{itemize}
            \item Increase $b_S$ so that $dC/db_S * \Delta b_S^* = C(b^c,b^a)$
            \item Decrease $b_L$ so that $dC/db_L * \Delta b_L^* = -C(b^c,b^a)$
        \end{itemize}
        \item Estimate $dC/db_S$ ($dC/db_L$) with $C(b,b_S)/\Delta b_S$ ($C(b,b_L)/\Delta b_L$)
        \item Implementation advantage:
        \begin{itemize}
            \item Optimal pension \textbf{schedule} depends on sufficient statistics under the \textbf{optimal} schedule
            \item Optimal pension \textbf{reform} depends on sufficient statistics under the \textbf{observed} schedule
        \end{itemize}
    \end{itemize}

\end{frame}
